```javascript
    showToast('操作成功！🎉', 'success');
}, 1000);

setTimeout(() => {
    console.log('测试3：显示错误提示');

}, 2000);

    console.log('✅ 测试完成！如果看到3个不同颜色的提示，说明功能正常');
}, 5000);
```
